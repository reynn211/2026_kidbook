# 2. Поиск надклассов (superclass) для наших концептов (чтобы понять иерархию)
SELECT ?item ?itemLabel ?superclass ?superclassLabel WHERE {
  VALUES ?item {
    wd:Q485        # Компьютерный вирус
    wd:Q93249      # Антивирусная программа
    wd:Q135005     # Фишинг
    wd:Q1487       # Хакер
  }
  OPTIONAL { ?item wdt:P279 ?superclass. }   # P279 = подкласс (subclass of)
  SERVICE wikibase:label { bd:serviceParam wikibase:language "ru". }
}

# 3. Найти все виды вредоносных программ (подклассы Malware - Q14001)
SELECT ?malware ?malwareLabel WHERE {
  ?malware wdt:P279 wd:Q14001.      # подкласс вредоносного ПО (Malware)
  SERVICE wikibase:label { bd:serviceParam wikibase:language "ru". }
}
LIMIT 20

# 4. Найти известные антивирусные программы (экземпляры класса Антивирус - Q93249)
SELECT ?antivirus ?antivirusLabel ?developer ?developerLabel WHERE {
  ?antivirus wdt:P31 wd:Q93249.     # экземпляр антивируса
  OPTIONAL { ?antivirus wdt:P178 ?developer. } # разработчик
  SERVICE wikibase:label { bd:serviceParam wikibase:language "ru". }
}
LIMIT 20
